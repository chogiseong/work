void
ProvisioningTableConsumer::checkRegion(const std::string &region, const std::string &regionDB)
{
    MSG_HIGH("checkRegion region: %s, regionDB: %s" , region.c_str(), regionDB.c_str());
    if(regionDB.empty())
    {
        MSG_HIGH("DB region value is empty");
        saveRegion(region);
    }
    else if(region != regionDB)
    {
        MSG_HIGH("received region value and DB region value are not equal");
        deleteConfAndDB();
        saveRegion(region);
    }
    else
    {
        MSG_HIGH("received region value and DB region value are equal");
    }
}

void
ProvisioningTableConsumer::deleteConfAndDB()
{
    MSG_HIGH("deleteConfAndDB start");
    std::remove("/appdata/.tele/db/TeleManager.db");
    std::system("rm -rf /appdata/.tele/.conf/*");
    //std::system("rm -rf /appdata/.tele/db/TeleManager.db");
}

void
ProvisioningTableConsumer::saveRegion(const std::string &region)
{
    MSG_HIGH("saveRegion start");
    DataBuffer::instance().setProvisioningData("Vehicle", "Region", region);
    //DataBuffer::instance().setRegionFromDB(region);
}
